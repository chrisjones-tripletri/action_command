Overview

A static description of the input and output from a given command. Although adding this adds a bunch of documentation and validation, it is not required. If you don't want to specify your input and output, you can just access the hash you passed into the command as @params

def initialize(action, desc)
  @action = action
  @desc = desc
  @params = []

  # universal parameters.
  param(:help, 'Help on this command', OPTIONAL)
  param(:rspec, 'Optional rspec context for performing validations via rspec_validate', OPTIONAL)
  param(:parent, 'Reference to the parent of this command, a symbol at the root', OPTIONAL)
end

def assign_args(dest, args)
  # handle aliasing
  args = preprocess_args(args)

  if(validate(args))
    @params.each do |param|
      sym = param[:symbol]
      if(args.has_key?(sym))
        sym_assign = "#{sym}=".to_sym
        dest.send(sym_assign, args[sym])      
      end
    end
  end
end

def is_help?(args)
  first_arg_sym = @params.first[:symbol]
  first_arg_val = args[first_arg_sym]
  return first_arg_val == 'help'
end

def show_help
  puts "#{@action.name}: #{@desc}"
  @params.each do |p|
    puts "    #{p[:symbol]}: #{p[:desc]}"
  end
end

def validate(args)
  @params.each do |p|
    val = args[p[:symbol]]
    if(!val || val == '*' || val == '')
      opts = p[:opts]
      if(!opts[:optional])
        puts "ERROR: * cannot be used for required param #{p[:symbol]}"
        return false
      end
    end
  end
  return true
end
